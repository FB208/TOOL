 public static byte[] GetPhoto(HttpPostedFile hpf)
    {
        int fileLen = hpf.ContentLength;

        Byte[] fileData = new Byte[fileLen];

        Stream sr = hpf.InputStream;

        sr.Read(fileData, 0, fileLen);

        return fileData;

    }
